1. 计算结构体成员的偏移量
```c
#define offsetof(TYPE,MEMBER)   (size_t) &((TYPE *)0)->MEMBER)
```
2. 计算字节中`bit`为`1`的数量
``` c
int Count(BYTE v)
{
    int num = 0;
    while (v)
    {
        v &= (v - 1); //v=v&(v-1)这个操作可以直接消除掉v中的最右边的1。
        num++;
    }
    return num;
}
```
3. 将字节中`1`分量分离并打印
```c
void print_one_bit(uint16 v)
{
    uint16 prev = 0;
    prev = v;
    while (v)
    {
        v &= (v - 1);
        printf("0x%04x\n", v ^ prev);
        prev = v;
    }
}
```
4. 判断大小端
- 大端，数据的高字节保存在内存的低地址处，数据的低字节保存在内存的高地址处
- 小端，数据的高字节保存在内存的高地址处，数据的低字节保存在内存的低地址处
``` c++
#include <iostream>
using namespace std;
int main()
{
    int a = 1;
    if (*(char*)&a == 1)
        cout << "小端模式" << endl;
    else
        cout << "大端模式" << endl;
    return 0;
}
```
